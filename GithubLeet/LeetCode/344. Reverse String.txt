
class Solution
{
    public void reverseString(char[] s)
    {
        
       int length = s.length;
       int half = s.length/2;
       for(int i = 0; i<half; i++)
       {
           char temp = s[i];
           s[i]= s[length-i-1];
           s[length-i-1] = temp;
       } 
      System.out.println(s);
    }
}
